```swift
func reverseList(_ head: ListNode?) -> ListNode? {
    guard let node = head else { return nil }

    var cur: ListNode? = node
    var pre: ListNode? = nil

    while cur != nil {
        let next = cur?.next
        cur?.next = pre
        pre = cur
        cur = next
    }

    return pre
}
```

### 求倒数第K个节点

```swift
func getKthNodeBackforward(_ node: ListNode?, _ k: Int) -> ListNode? {
    guard let node = node else { return nil }
    
    var p1: ListNode? = node
    var p2: ListNode? = node
    
    var i = 0
    
    while p1?.next != nil && i < k {
        p1 = p1?.next
        i += 1
    }
    
    if p1 == nil {
        return nil
    }
    
    while p1?.next != nil {
        p1 = p1?.next
        p2 = p2?.next
    }
    
    return p2
}
```

### 链表相交

```swift
func getIntersectionNode(_ headA: ListNode?, _ headB: ListNode?) -> ListNode? {
    var a = headA
    var b = headB

    while a !== b {
        a = a != nil ? a?.next : headB
        b = b != nil ? b?.next : headA
    }

    return a
}
```

### 带环链表

检测环，找环入口. 检测环也是快慢指针，当指针相遇时，不为null，则存在环；否则无环；

```swift
func detectCycle(_ head: ListNode?) -> ListNode? {
    guard let head = head else { return nil }

    if head.next == nil {
        return nil
    }

    var slow: ListNode? = head
    var fast: ListNode? = head

    while slow != nil && fast?.next != nil {
        slow = slow?.next
        fast = fast?.next?.next

	  // 环内相遇
        if slow === fast {
            break
        }
    }

    // 快指针从头走
    fast = head
    
    // 继续走直到相遇即是入口
    while slow !== fast {
        slow = slow?.next
        fast = fast?.next
    }
    
    return slow
}
```

### 链表相加 (两数相加)

```swift
func addTwoNumbers(_ l1: ListNode?, _ l2: ListNode?) -> ListNode? {
    var head1 = l1
    var head2 = l2
    var add = 0
    var dummy: ListNode = ListNode(-1)
    var cur = dummy

    while head1 != nil || head2 != nil {
        let a = head1?.val ?? 0
        let b = head2?.val ?? 0
        var newValue = a+b+add

        if newValue > 9 {
            newValue = newValue % 10
            add = 1
        } else {
            add = 0
        }

        var temp = ListNode(newValue)

        cur.next = temp
        cur = temp

        head1 = head1?.next
        head2 = head2?.next
    }

    if add == 1 {
        var temp = ListNode(1)
        cur.next = temp
    }

    return dummy.next
}
```

```swift
func mergeKLists(_ lists: [ListNode?]) -> ListNode? {
```

### 链表排序 (Leetcode-148)
理解Merge Sort

```swift
func sortList(_ head: ListNode?) -> ListNode? {
    if (head == nil || head?.next == nil) {
        return head
    }
    var slow = head
    var fast = head?.next

    // 快慢指针法，使得slow指向链表的中间节点。
    while(fast != nil && fast?.next != nil){
        slow = slow?.next
        fast = fast?.next?.next
    }

    let last = slow?.next
    slow?.next = nil
    let first = sortList(head)
    let second = sortList(last)
    let res = mergeTwoList(first,second)
    return res;
}

func mergeTwoList(_ list1: ListNode?, _ list2: ListNode?) -> ListNode? {
    let dummpy = ListNode(-1)
    var tail = dummpy
    var list1 = list1
    var list2 = list2

    while(list1 != nil && list2 != nil){
        if(list1?.val ?? 0 < list2?.val ?? 0){
            tail.next = list1
            list1 = list1?.next
        } else{
            tail.next = list2
            list2 = list2?.next
        }
        tail = tail.next!
    }

    tail.next = (list1 != nil ? list1 : list2);
    return dummpy.next
}
```

### LRU (Leetcode-146)
注意每个方法的时间复杂度.

```swift
final class LRUCache {
    // Swift 需要自己实现双向链表，或者队列
    private(set) var capacity: Int = 0
    private(set) var dictionary = [Int: Node]()
    var first: Node!
    var last: Node!
    
    init(_ capacity: Int) {
        self.capacity = capacity
        dictionary = [Int:Node]()
        
        first = Node()
        last = Node()
        
        first.next = last
        last.prev = first
    }
    
    // tc: O(1)
    func get(_ key: Int) -> Int {
        guard let node = dictionary[key] else {
            return -1
        }
        
        removeNode(node)
        addAtFirst(node)
        
        return node.val
    }
    
    // tc: O(n)
    func put(_ key: Int, _ value: Int) {
        let node = dictionary[key]
        if node != nil {
            //! 更新 key
            node!.val = value
            removeNode(node!)
            addAtFirst(node!)
        } else {
            if dictionary.keys.count == capacity {
                //! 淘汰最近最少使用
                let prevNode = dictionary.removeValue(forKey: last.prev!.key)!   // O(n)
                removeNode(prevNode)
            }
            
            let newNode = Node(key, value)
            dictionary[key] = newNode
            addAtFirst(newNode)
        }
    }
    
    //! 双向链表 -> 删除 自身
    private func removeNode(_ node: Node) {
        node.next!.prev = node.prev
        node.prev!.next = node.next
    }
    
    //! 双向链表 -> 插入节点到头节点后面
    private func addAtFirst(_ node: Node) {
        node.next = first.next
        first.next!.prev = node
        
        first.next = node
        node.prev = first
    }
}

class Node {
    var key:Int = 0
    var val: Int = 0
    
    var prev: Node?
    var next: Node?
    
    init(_ key: Int = 0, _ value: Int = 0) {
        self.key = key
        self.val = value
    }
}
```

### 复制带随机指针的链表 (Leetcode-138)
对于链表复制， 要熟练使用hashmap做映射; 若想空间复杂度为O(1), 则可以在原链表上操作，给每个node添加一个而外的next node，为它的副本.

```swift
var hashMap = [Node: Node]()
func copyRandomList(_ head: Node?) -> Node? {
    if head == nil {
        return nil
    }

    if let areadyNode = hashMap[head!] {
        return areadyNode
    }

    let createNode = Node(head!.val)
    hashMap[head!] = createNode
    createNode.next = copyRandomList(head?.next)
    createNode.random = copyRandomList(head?.random)

    return hashMap[head!]
}
```

### 填充每个节点的下一个右侧节点指针 (Leetcode-116)
针对next指针发力，next指针是横向指针，我们可以利用两层循环来模拟递归：第一层是纵向指针，遍历每层开头节点；第二层是横向指针，遍历每层:

```swift
 func connect(_ root: Node?) -> Node? {
    guard var _ = root else { return nil }

    var node: Node? = root
    
    // 外层循环控制纵向遍历
    while node != nil {
        var temp: Node? = node
        
        // 内层循环控制横向遍历
        while temp != nil {
            temp!.left?.next = temp!.right
            
            if temp!.next != nil {
                temp!.right?.next = temp!.next?.left
            }
            
            temp = temp!.next
        }
        
        node = node?.left
    }
    
    return root
}
```

###  扁平化多级双向链表 (Leetcode-430)
该题指针众多，还是双向链表，别忘了每个指针都要更新:

```swift
func flatten(_ head: Node?) -> Node? {
    guard var _ = head else { return nil }

    var node: Node? = head
    let _ = mergeChild(node)

    return node
}

func mergeChild(_ node: Node?) -> (Node?, Node?) {
    var node: Node? = node
    var start = node
    var end = node

    while node != nil {
        if node?.child == nil {
            node = node?.next
        } else {
            let (begin, last) = mergeChild(node?.child)

            let temp = node!.next
            node?.child = nil
            node!.next = begin
            begin?.prev = node
            temp?.prev = last
            last?.next = temp

            node = node?.next
        }

        if node != nil {
            end = node
        }
    }

    return (start, end)
}
```
